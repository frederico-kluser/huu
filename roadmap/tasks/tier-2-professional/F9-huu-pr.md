### Code sketch (`src/lib/pr-body-builder.ts`)

```typescript
import type { GateReport } from '../janitor/gate-types.js';
import type { ForecastReport } from '../forecast/index.js';

interface RunSummary {
  runId: string;
  pipelineName: string;
  pipelinePath: string;
  stagesCount: number;
  tasksCount: number;
  durationMs: number;
  totalCostUsd: number;
  forecast?: ForecastReport;
  gateReport?: GateReport;
  auditValid?: boolean;
}

export function buildPrBody(s: RunSummary): string {
  const lines: string[] = [];
  lines.push('## huu run summary');
  lines.push('');
  lines.push(`Pipeline: \`${s.pipelinePath}\``);
  lines.push(`Stages: ${s.stagesCount} · Tasks: ${s.tasksCount} · Duration: ${formatDuration(s.durationMs)}`);
  lines.push(`Cost: $${s.totalCostUsd.toFixed(2)}` + (s.forecast ? `  (forecast was $${s.forecast.totalCostUsd.toFixed(2)})` : ''));
  lines.push('');

  if (s.gateReport && s.gateReport.gates.length > 0) {
    lines.push('## Quality gates');
    lines.push('');
    for (const g of s.gateReport.gates) {
      const sym = g.passed ? '✓' : '✗';
      lines.push(`${sym} ${g.name.padEnd(8)} ${formatDuration(g.durationMs)}`);
    }
    if (!s.gateReport.allPassed) {
      lines.push('');
      lines.push('⚠ Some required gates failed. Review details above.');
    }
    lines.push('');
  }

  lines.push('## Audit');
  lines.push('');
  lines.push(`Run ID: \`${s.runId}\``);
  if (s.auditValid !== undefined) {
    lines.push(s.auditValid
      ? `Audit: ✓ HMAC chain valid (verify with \`huu audit verify ${s.runId}\`)`
      : `Audit: ✗ HMAC chain BROKEN — investigate immediately`);
  }
  lines.push('');
  lines.push('🤖 Generated by [huu](https://github.com/frederico-kluser/huu)');
  return lines.join('\n');
}

function formatDuration(ms: number): string {
  if (ms < 1000) return `${ms}ms`;
  if (ms < 60_000) return `${(ms / 1000).toFixed(1)}s`;
  return `${Math.floor(ms / 60_000)}m ${Math.floor((ms % 60_000) / 1000)}s`;
}
```

### Code sketch (`src/cli/commands/pr.ts`)

```typescript
import { execa } from 'execa';
import * as fs from 'node:fs/promises';
import * as path from 'node:path';
import { resolveRepoRoot } from '../../git/git-client.js';
import { buildPrBody } from '../../lib/pr-body-builder.js';
import { AuditLog } from '../../audit/audit-log.js';

export async function runPrCommand(argv: string[]): Promise<number> {
  const repoRoot = resolveRepoRoot(process.cwd());
  const draft = argv.includes('--draft');
  const reviewerArg = argv[argv.indexOf('--reviewer') + 1];
  const runId = argv.find((a) => !a.startsWith('--')) ?? await findLatestRunId(repoRoot);

  if (!runId) {
    console.error('No runId provided and no recent runs found.');
    return 2;
  }

  const summary = await loadRunSummary(repoRoot, runId);
  if (!summary) {
    console.error(`Run not found: ${runId}`);
    return 1;
  }

  // Verify audit
  try {
    const audit = await AuditLog.verify(repoRoot, runId);
    summary.auditValid = audit.valid;
  } catch {
    summary.auditValid = undefined;
  }

  const body = buildPrBody(summary);
  const title = `huu: ${summary.pipelineName}`;

  // Push current branch first
  const { stdout: branchName } = await execa('git', ['rev-parse', '--abbrev-ref', 'HEAD'], { cwd: repoRoot });
  await execa('git', ['push', '-u', 'origin', branchName.trim()], { cwd: repoRoot, stdio: 'inherit' });

  // Try gh CLI first
  const args = ['pr', 'create', '--title', title, '--body', body];
  if (draft) args.push('--draft');
  if (reviewerArg) args.push('--reviewer', reviewerArg);

  try {
    const { stdout } = await execa('gh', args, { cwd: repoRoot });
    console.log(stdout);
    return 0;
  } catch (err) {
    // Fallback to GitHub API via fetch + GITHUB_TOKEN
    return fallbackToApi(repoRoot, branchName.trim(), title, body);
  }
}

async function findLatestRunId(repoRoot: string): Promise<string | null> {
  const dir = path.join(repoRoot, '.huu', 'audit');
  try {
    const files = await fs.readdir(dir);
    const runs = files.filter((f) => f.endsWith('.jsonl')).map((f) => f.replace('.jsonl', ''));
    return runs.sort().at(-1) ?? null;
  } catch { return null; }
}

async function loadRunSummary(repoRoot: string, runId: string): Promise<any> {
  // Read from `.huu/<runId>-summary.json` written by orchestrator at run end.
  // Schema: { runId, pipelineName, pipelinePath, stagesCount, tasksCount, durationMs, totalCostUsd, forecast?, gateReport? }
  try {
    const raw = await fs.readFile(path.join(repoRoot, '.huu', `${runId}-summary.json`), 'utf-8');
    return JSON.parse(raw);
  } catch {
    return null;
  }
}

async function fallbackToApi(repoRoot: string, branch: string, title: string, body: string): Promise<number> {
  const token = process.env.GITHUB_TOKEN;
  if (!token) {
    console.error('gh CLI not found and GITHUB_TOKEN not set.');
    return 1;
  }
  // ... implement minimal POST to /repos/:owner/:repo/pulls ...
  console.error('GITHUB_TOKEN fallback not yet implemented; install gh CLI.');
  return 1;
}
```

### Wire orchestrator to write summary

After run finishes, persist `.huu/<runId>-summary.json` with the data
needed by `buildPrBody`. Done by orchestrator subscriber that listens
to `run_finished`.
