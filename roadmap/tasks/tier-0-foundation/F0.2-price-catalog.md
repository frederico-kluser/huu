## Context

`huu` calcula custo *post-hoc* hoje: depois que a Pi SDK retorna,
`usage.cost.total` é gravado por agente (`src/orchestrator/real-agent.ts`).
Não conseguimos estimar custo *antes* de uma run.

A feature **F2 (`huu run --dry-run`)** depende de saber preço por
modelo + tokens estimados. Sem catálogo de preços, "estimativa" reduz a
"peça pro usuário inserir o preço" — UX fraca, e usuário comum não sabe.

OpenRouter expõe `https://openrouter.ai/api/v1/models` (público, sem auth)
com preços por modelo em `pricing.{prompt,completion,request,image}`. Esta
task constrói o catálogo + cache + fallback.

### Code sketch (`src/lib/price-catalog.ts`)

```typescript
import * as fs from 'node:fs/promises';
import * as path from 'node:path';
import * as os from 'node:os';

const CACHE_DIR = path.join(os.homedir(), '.huu');
const CACHE_FILE = path.join(CACHE_DIR, 'prices.json');
const TTL_MS = 24 * 60 * 60 * 1000; // 24h
const FETCH_TIMEOUT_MS = 5_000;
const OPENROUTER_URL = 'https://openrouter.ai/api/v1/models';

export interface ModelPrice {
  /** USD per 1M input tokens. */
  inputPerMillion: number;
  /** USD per 1M output tokens. */
  outputPerMillion: number;
  /** USD per 1M cached input tokens (read). 0 if not supported. */
  cacheReadPerMillion: number;
  /** USD per 1M cached input tokens (write/breakpoint). 0 if not supported. */
  cacheWritePerMillion: number;
  /** Approx max output tokens per turn (used for estimation prior). */
  maxOutputTokens?: number;
}

export interface PriceCatalog {
  fetchedAt: number;
  source: 'live' | 'cache' | 'fallback';
  prices: Record<string, ModelPrice>;
}

interface OpenRouterModelEntry {
  id: string;
  pricing: {
    prompt: string;       // USD per token, as string (e.g. "0.000003")
    completion: string;
    request?: string;
    image?: string;
    input_cache_read?: string;
    input_cache_write?: string;
  };
  top_provider?: { max_completion_tokens?: number };
}

async function fetchLive(timeoutMs = FETCH_TIMEOUT_MS): Promise<PriceCatalog> {
  const ctrl = new AbortController();
  const t = setTimeout(() => ctrl.abort(), timeoutMs);
  try {
    const res = await fetch(OPENROUTER_URL, { signal: ctrl.signal });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const body = (await res.json()) as { data: OpenRouterModelEntry[] };
    const prices: Record<string, ModelPrice> = {};
    for (const m of body.data) {
      prices[m.id] = {
        inputPerMillion: parseFloat(m.pricing.prompt) * 1_000_000,
        outputPerMillion: parseFloat(m.pricing.completion) * 1_000_000,
        cacheReadPerMillion: parseFloat(m.pricing.input_cache_read ?? '0') * 1_000_000,
        cacheWritePerMillion: parseFloat(m.pricing.input_cache_write ?? '0') * 1_000_000,
        maxOutputTokens: m.top_provider?.max_completion_tokens,
      };
    }
    return { fetchedAt: Date.now(), source: 'live', prices };
  } finally {
    clearTimeout(t);
  }
}

async function readCache(): Promise<PriceCatalog | null> {
  try {
    const raw = await fs.readFile(CACHE_FILE, 'utf-8');
    const parsed = JSON.parse(raw) as PriceCatalog;
    if (Date.now() - parsed.fetchedAt < TTL_MS) {
      return { ...parsed, source: 'cache' };
    }
    return null; // stale
  } catch {
    return null;
  }
}

async function writeCache(catalog: PriceCatalog): Promise<void> {
  await fs.mkdir(CACHE_DIR, { recursive: true });
  await fs.writeFile(CACHE_FILE, JSON.stringify(catalog, null, 2), 'utf-8');
}

async function readFallback(): Promise<PriceCatalog> {
  // path resolved relative to compiled dist/ — adjust based on tsconfig:
  const fallbackPath = path.resolve(__dirname, '../../data/openrouter-prices-fallback.json');
  const raw = await fs.readFile(fallbackPath, 'utf-8');
  const prices = JSON.parse(raw) as Record<string, ModelPrice>;
  return { fetchedAt: 0, source: 'fallback', prices };
}

/**
 * Returns a price catalog. Order of preference: cache (fresh) → live fetch → fallback.
 * Never throws — fallback is always shipped in the repo.
 */
export async function getPriceCatalog(opts: {
  forceRefresh?: boolean;
  offline?: boolean;
} = {}): Promise<PriceCatalog> {
  if (!opts.forceRefresh) {
    const cached = await readCache();
    if (cached) return cached;
  }
  if (!opts.offline) {
    try {
      const live = await fetchLive();
      await writeCache(live);
      return live;
    } catch {
      // fall through to fallback
    }
  }
  return readFallback();
}

export interface CostEstimate {
  usd: number;
  /** 95% confidence interval delta (±). */
  ci95: number;
  modelId: string;
  modelKnown: boolean;
}

export function estimateCost(
  catalog: PriceCatalog,
  modelId: string,
  tokensIn: number,
  tokensOut: number,
  cacheReadTokens = 0,
  cacheWriteTokens = 0,
): CostEstimate {
  const p = catalog.prices[modelId];
  if (!p) {
    return { usd: 0, ci95: 0, modelId, modelKnown: false };
  }
  const usd =
    (tokensIn / 1_000_000) * p.inputPerMillion +
    (tokensOut / 1_000_000) * p.outputPerMillion +
    (cacheReadTokens / 1_000_000) * p.cacheReadPerMillion +
    (cacheWriteTokens / 1_000_000) * p.cacheWritePerMillion;
  // Naive 30% CI for now; F2 may refine using F0.3 history.
  return { usd, ci95: usd * 0.3, modelId, modelKnown: true };
}
```

### Code sketch (`src/cli/commands/prices.ts`)

```typescript
import { getPriceCatalog } from '../../lib/price-catalog.js';

export async function runPricesCommand(argv: string[]): Promise<number> {
  const refresh = argv.includes('--refresh') || argv.includes('-r');
  const json = argv.includes('--json');
  const offline = argv.includes('--offline');

  const catalog = await getPriceCatalog({ forceRefresh: refresh, offline });
  if (json) {
    process.stdout.write(JSON.stringify(catalog, null, 2) + '\n');
    return 0;
  }
  console.log(`Source: ${catalog.source}`);
  console.log(`Fetched: ${new Date(catalog.fetchedAt).toISOString()}`);
  console.log(`Models: ${Object.keys(catalog.prices).length}`);
  // Compact table:
  const top = Object.entries(catalog.prices)
    .sort((a, b) => b[1].outputPerMillion - a[1].outputPerMillion)
    .slice(0, 20);
  for (const [id, p] of top) {
    console.log(`  ${id.padEnd(40)} in=$${p.inputPerMillion.toFixed(2)}/M  out=$${p.outputPerMillion.toFixed(2)}/M`);
  }
  return 0;
}
```

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest';
import * as fs from 'node:fs/promises';
import { getPriceCatalog, estimateCost } from './price-catalog.js';

beforeEach(() => {
  vi.restoreAllMocks();
});

describe('estimateCost', () => {
  const catalog = {
    fetchedAt: Date.now(),
    source: 'live' as const,
    prices: {
      'anthropic/claude-sonnet-4-6': {
        inputPerMillion: 3, outputPerMillion: 15,
        cacheReadPerMillion: 0.3, cacheWritePerMillion: 3.75,
      },
    },
  };

  it('computes USD from tokens', () => {
    const e = estimateCost(catalog, 'anthropic/claude-sonnet-4-6', 1_000_000, 100_000);
    // 1M * 3 + 0.1M * 15 = 3 + 1.5 = $4.50
    expect(e.usd).toBeCloseTo(4.5, 2);
    expect(e.modelKnown).toBe(true);
  });

  it('returns modelKnown=false for unknown id', () => {
    const e = estimateCost(catalog, 'foo/bar', 1000, 1000);
    expect(e.modelKnown).toBe(false);
    expect(e.usd).toBe(0);
  });
});

describe('getPriceCatalog', () => {
  it('falls back to bundled JSON when fetch fails and no cache', async () => {
    vi.spyOn(global, 'fetch').mockRejectedValue(new Error('network'));
    vi.spyOn(fs, 'readFile').mockImplementation(async (p) => {
      if (String(p).includes('prices.json')) throw Object.assign(new Error('ENOENT'), { code: 'ENOENT' });
      return '{"some/model": {"inputPerMillion":1,"outputPerMillion":2,"cacheReadPerMillion":0,"cacheWritePerMillion":0}}';
    });
    const c = await getPriceCatalog();
    expect(c.source).toBe('fallback');
    expect(c.prices['some/model']).toBeDefined();
  });
});
```
