### Code sketch (`src/lib/history.ts`)

```typescript
import * as fs from 'node:fs/promises';
import * as fsSync from 'node:fs';
import * as path from 'node:path';
import { createGzip } from 'node:zlib';
import { pipeline } from 'node:stream/promises';
import { createReadStream, createWriteStream } from 'node:fs';

const ROTATION_BYTES = 10 * 1024 * 1024; // 10 MB
const HISTORY_FILE = '.huu/history.jsonl';

export interface HistoryEntry {
  /** ISO timestamp. */
  ts: string;
  runId: string;
  stageIndex: number;
  stepName: string;
  agentId: number;
  modelId: string;
  /** Files this agent touched (or [] for whole-project). */
  files: string[];
  success: boolean;
  errorKind?: 'timeout' | 'failed';
  costUsd: number;
  tokensIn: number;
  tokensOut: number;
  cacheReadTokens: number;
  cacheWriteTokens: number;
  durationMs: number;
  /** F1 results when available; undefined until F1 lands. */
  gateFailures?: string[];
}

function historyPath(repoRoot: string): string {
  return path.join(repoRoot, HISTORY_FILE);
}

async function rotateIfNeeded(absPath: string): Promise<void> {
  let stat;
  try {
    stat = await fs.stat(absPath);
  } catch {
    return; // file doesn't exist yet — no rotation
  }
  if (stat.size < ROTATION_BYTES) return;

  // Find next .N.gz suffix
  let n = 1;
  while (fsSync.existsSync(`${absPath}.${n}.gz`)) n++;
  const target = `${absPath}.${n}.gz`;

  // Rename then gzip the old file (atomic-ish on the same FS)
  const tmp = `${absPath}.rotating`;
  await fs.rename(absPath, tmp);
  await pipeline(createReadStream(tmp), createGzip(), createWriteStream(target));
  await fs.unlink(tmp);
}

/** Append a single history entry. Concurrency-safe via O_APPEND. */
export async function appendHistory(repoRoot: string, entry: HistoryEntry): Promise<void> {
  const abs = historyPath(repoRoot);
  await fs.mkdir(path.dirname(abs), { recursive: true });
  await rotateIfNeeded(abs);
  const line = JSON.stringify(entry) + '\n';
  // O_APPEND ensures atomic append across concurrent agents (POSIX guarantee for <PIPE_BUF=4096 bytes).
  // Each entry is well under 4096 bytes. Use lower-level `fsSync` + flag 'a' to be safe.
  await fs.appendFile(abs, line, { encoding: 'utf-8', flag: 'a' });
}

export interface HistoryFilter {
  runId?: string;
  modelId?: string;
  stepName?: string;
  successOnly?: boolean;
  failureOnly?: boolean;
  /** Only entries newer than this ms-since-epoch. */
  since?: number;
}

/** Query the live (uncompressed) history file. Rotated archives are NOT scanned. */
export async function queryHistory(
  repoRoot: string,
  filter: HistoryFilter = {},
): Promise<HistoryEntry[]> {
  const abs = historyPath(repoRoot);
  let raw: string;
  try {
    raw = await fs.readFile(abs, 'utf-8');
  } catch {
    return [];
  }
  const out: HistoryEntry[] = [];
  for (const line of raw.split('\n')) {
    if (!line.trim()) continue;
    try {
      const entry = JSON.parse(line) as HistoryEntry;
      if (filter.runId && entry.runId !== filter.runId) continue;
      if (filter.modelId && entry.modelId !== filter.modelId) continue;
      if (filter.stepName && entry.stepName !== filter.stepName) continue;
      if (filter.successOnly && !entry.success) continue;
      if (filter.failureOnly && entry.success) continue;
      if (filter.since && new Date(entry.ts).getTime() < filter.since) continue;
      out.push(entry);
    } catch {
      // skip malformed lines silently — log corruption shouldn't crash queries
    }
  }
  return out;
}

/** Aggregate stats useful for F6-lite advisor. */
export interface HistoryStats {
  count: number;
  successRate: number;
  avgCostUsd: number;
  avgDurationMs: number;
  p50CostUsd: number;
  p95CostUsd: number;
}

export function aggregate(entries: HistoryEntry[]): HistoryStats {
  if (entries.length === 0) {
    return { count: 0, successRate: 0, avgCostUsd: 0, avgDurationMs: 0, p50CostUsd: 0, p95CostUsd: 0 };
  }
  const costs = entries.map((e) => e.costUsd).sort((a, b) => a - b);
  const successes = entries.filter((e) => e.success).length;
  return {
    count: entries.length,
    successRate: successes / entries.length,
    avgCostUsd: costs.reduce((a, b) => a + b, 0) / entries.length,
    avgDurationMs: entries.reduce((a, e) => a + e.durationMs, 0) / entries.length,
    p50CostUsd: costs[Math.floor(costs.length * 0.5)],
    p95CostUsd: costs[Math.floor(costs.length * 0.95)],
  };
}
```

## Tests

### Unit (`src/lib/history.test.ts`)

```typescript
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import * as fs from 'node:fs/promises';
import * as path from 'node:path';
import * as os from 'node:os';
import { appendHistory, queryHistory, aggregate, type HistoryEntry } from './history.js';

let tmpRoot: string;

beforeEach(async () => {
  tmpRoot = await fs.mkdtemp(path.join(os.tmpdir(), 'huu-history-'));
});
afterEach(async () => {
  await fs.rm(tmpRoot, { recursive: true, force: true });
});

const sample = (overrides: Partial<HistoryEntry> = {}): HistoryEntry => ({
  ts: new Date().toISOString(),
  runId: 'run-1',
  stageIndex: 0,
  stepName: 'step-A',
  agentId: 1,
  modelId: 'anthropic/claude-sonnet-4-6',
  files: ['src/foo.ts'],
  success: true,
  costUsd: 0.05,
  tokensIn: 1000,
  tokensOut: 500,
  cacheReadTokens: 0,
  cacheWriteTokens: 0,
  durationMs: 1234,
  ...overrides,
});

describe('appendHistory + queryHistory', () => {
  it('round-trips a single entry', async () => {
    await appendHistory(tmpRoot, sample());
    const all = await queryHistory(tmpRoot);
    expect(all).toHaveLength(1);
    expect(all[0].stepName).toBe('step-A');
  });

  it('filters by modelId', async () => {
    await appendHistory(tmpRoot, sample({ modelId: 'a/b' }));
    await appendHistory(tmpRoot, sample({ modelId: 'c/d' }));
    const filtered = await queryHistory(tmpRoot, { modelId: 'a/b' });
    expect(filtered).toHaveLength(1);
  });

  it('survives concurrent appends', async () => {
    await Promise.all(
      Array.from({ length: 100 }, (_, i) =>
        appendHistory(tmpRoot, sample({ agentId: i })),
      ),
    );
    const all = await queryHistory(tmpRoot);
    expect(all).toHaveLength(100);
  });

  it('returns [] for missing file', async () => {
    const all = await queryHistory(tmpRoot);
    expect(all).toEqual([]);
  });
});

describe('aggregate', () => {
  it('computes p50/p95 and success rate', () => {
    const entries = Array.from({ length: 100 }, (_, i) =>
      sample({ costUsd: i / 100, success: i < 80 }),
    );
    const stats = aggregate(entries);
    expect(stats.successRate).toBe(0.8);
    expect(stats.p50CostUsd).toBeCloseTo(0.5, 1);
    expect(stats.p95CostUsd).toBeCloseTo(0.95, 1);
  });
});
```
