## Context

`huu` é um orquestrador TypeScript/Ink que executa pipelines paralelos. A
pipeline é um JSON portátil (`huu-pipeline-v1`) — formato versionado que vai
para gist, repo, ou cookbook. Hoje o schema vive como `interface` puros em
`src/lib/types.ts`, e a validação de pipelines carregadas é feita por
checagens manuais ad-hoc em `src/lib/pipeline-io.ts`.

**Problema:** features futuras (F1 quality gates, F2 dry-run cost, F3 MCP
server, F8 JSON Schema/LSP) precisam todas validar pipelines vindas de
fontes externas. Sem fonte única, vamos manter 4 representações divergentes
do mesmo schema — divergências viram bugs.

**Solução:** adotar `zod` como única fonte. Tipos TypeScript derivam via
`z.infer`. JSON Schema (para SchemaStore/LSP) deriva via `zod-to-json-schema`.
Todo I/O de pipeline passa por `pipelineSchema.parse()`.

### Code sketch (`src/schema/pipeline-v1.ts`)

```typescript
import { z } from 'zod';

// --- Sub-schemas ---

export const stepScopeSchema = z.enum(['project', 'per-file', 'flexible']);
export type StepScope = z.infer<typeof stepScopeSchema>;

export const portAllocationConfigSchema = z.object({
  basePort: z.number().int().min(1024).max(65535).optional()
    .describe('First port in the allocation range. Default 55100.'),
  windowSize: z.number().int().min(10).max(1000).optional()
    .describe('Ports per agent. Min/default 10 (http, db, ws + 7 extras).'),
  enabled: z.boolean().optional()
    .describe('Set false to skip env-file generation entirely. Default true.'),
}).strict();

export type PortAllocationConfig = z.infer<typeof portAllocationConfigSchema>;

export const promptStepSchema = z.object({
  name: z.string().min(1, 'Step name cannot be empty'),
  prompt: z.string().min(1),
  files: z.array(z.string()).default([])
    .describe('Repo-relative paths. Empty = whole-project task.'),
  modelId: z.string().optional()
    .describe('Per-step model override. Falls back to AppConfig.modelId.'),
  scope: stepScopeSchema.optional()
    .describe('Decomposition. Undefined = flexible (back-compat).'),
  interactive: z.boolean().optional()
    .describe('Open multi-turn refinement chat before this stage.'),
  refinementModel: z.string().optional()
    .describe('Defaults to moonshotai/kimi-k2.6 when interactive is true.'),
}).strict()
  .refine(
    (step) => !(step.scope === 'per-file' && step.files.length === 0),
    { message: 'scope=per-file requires non-empty files', path: ['files'] },
  );

export type PromptStep = z.infer<typeof promptStepSchema>;

export const pipelineSchema = z.object({
  name: z.string().min(1),
  steps: z.array(promptStepSchema).min(1, 'Pipeline must have at least one step'),
  cardTimeoutMs: z.number().int().positive().optional(),
  singleFileCardTimeoutMs: z.number().int().positive().optional(),
  maxRetries: z.number().int().min(0).max(10).optional(),
  portAllocation: portAllocationConfigSchema.optional(),
}).strict();

export type Pipeline = z.infer<typeof pipelineSchema>;

// --- Wrapper format (huu-pipeline-v1.json on disk) ---

export const pipelineFileSchema = z.object({
  _format: z.literal('huu-pipeline-v1'),
  exportedAt: z.string().datetime().optional(),
  pipeline: pipelineSchema,
}).strict();

export type PipelineFile = z.infer<typeof pipelineFileSchema>;

// --- Defaults (replaces constants from src/lib/types.ts) ---

export const DEFAULT_CARD_TIMEOUT_MS = 600_000;
export const DEFAULT_SINGLE_FILE_CARD_TIMEOUT_MS = 300_000;
export const DEFAULT_MAX_RETRIES = 1;
export const DEFAULT_PORT_BASE = 55_100;
export const DEFAULT_PORT_WINDOW = 10;

// --- Helper: parse with friendly errors ---

export function parsePipelineFile(json: unknown): PipelineFile {
  const result = pipelineFileSchema.safeParse(json);
  if (!result.success) {
    const issues = result.error.issues
      .map((i) => `  • ${i.path.join('.')}: ${i.message}`)
      .join('\n');
    throw new Error(`Invalid huu-pipeline-v1 JSON:\n${issues}`);
  }
  return result.data;
}
```

### Code sketch (refactor `src/lib/pipeline-io.ts`)

```typescript
import { parsePipelineFile, type PipelineFile, type Pipeline } from '../schema/pipeline-v1.js';
import * as fs from 'node:fs';

export function loadPipelineFromFile(absPath: string): Pipeline {
  const raw = fs.readFileSync(absPath, 'utf-8');
  const json = JSON.parse(raw); // SyntaxError bubble up — that's fine
  const file = parsePipelineFile(json);
  return file.pipeline;
}

export function savePipelineToFile(absPath: string, pipeline: Pipeline): void {
  const file: PipelineFile = {
    _format: 'huu-pipeline-v1',
    exportedAt: new Date().toISOString(),
    pipeline,
  };
  // Validate before writing (defense-in-depth):
  parsePipelineFile(file);
  fs.writeFileSync(absPath, JSON.stringify(file, null, 2) + '\n', 'utf-8');
}
```

## Tests

### Unit (`src/schema/pipeline-v1.test.ts`)

```typescript
import { describe, it, expect } from 'vitest';
import { pipelineSchema, parsePipelineFile, promptStepSchema } from './pipeline-v1.js';

describe('promptStepSchema', () => {
  it('accepts a minimal valid step', () => {
    const step = { name: 'X', prompt: 'do X' };
    const parsed = promptStepSchema.parse(step);
    expect(parsed.files).toEqual([]); // default applied
  });

  it('rejects empty step name', () => {
    expect(() => promptStepSchema.parse({ name: '', prompt: 'x' })).toThrow();
  });

  it('rejects per-file scope with empty files', () => {
    expect(() => promptStepSchema.parse({
      name: 'X', prompt: 'p', scope: 'per-file', files: [],
    })).toThrow(/scope=per-file requires/);
  });

  it('rejects extra fields (strict)', () => {
    expect(() => promptStepSchema.parse({
      name: 'X', prompt: 'p', __unknownField: 1,
    })).toThrow();
  });
});

describe('pipelineSchema', () => {
  it('accepts pipeline with one step', () => {
    const p = pipelineSchema.parse({
      name: 'demo', steps: [{ name: 'a', prompt: 'b' }],
    });
    expect(p.steps).toHaveLength(1);
  });

  it('rejects empty steps array', () => {
    expect(() => pipelineSchema.parse({ name: 'x', steps: [] })).toThrow(/at least one/);
  });
});

describe('parsePipelineFile', () => {
  it('reports JSON path in error message', () => {
    const bad = {
      _format: 'huu-pipeline-v1',
      pipeline: { name: 'x', steps: [{ name: '', prompt: 'p' }] },
    };
    expect(() => parsePipelineFile(bad)).toThrow(/pipeline\.steps\.0\.name/);
  });
});
```
